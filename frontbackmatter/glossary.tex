% ---- Glossary Entries ----

\longnewglossaryentry{distant supervision}{name=distant supervision}
{%
  Automatically labeling of training texts by leveraging an existing database of known facts~\citep{reschke2014event}.
}

\longnewglossaryentry{freebase}{name=Freebase}
{%
  An online collection of structured data proposed by~\citep{bollacker2008freebase}.
  It's API was retired on June 30, 2015 (Source: \url{https://developers.google.com/freebase/v1/}).
}


\longnewglossaryentry{aho-corasick algorithm}{name=Aho-Corasick algorithm}
{%
  An algorith proposed by \citet{aho1975efficient}. It allows an efficient matching of strings in a dictionary to a given input string using finite state machines.
}
